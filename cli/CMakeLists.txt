project(VSProjectMakerCLI) 

include( include.cmake )
include( ${CMAKE_SOURCE_DIR}/SABUtils/Project.cmake )

add_executable( VSProjectMakerCLI
                 ${project_SRCS} 
                 ${project_H} 
                 ${qtproject_SRCS} 
                 ${qtproject_QRC} 
                 ${qtproject_QRC_SRCS} 
                 ${qtproject_UIS_H} 
                 ${qtproject_MOC_SRCS} 
                 ${qtproject_H} 
                 ${qtproject_UIS}
                 ${qtproject_QRC_SOURCES}
                 ${_CMAKE_FILES}
                 ${_CMAKE_MODULE_FILES}
                application.rc
          )
set_target_properties( VSProjectMakerCLI PROPERTIES FOLDER Apps )
          
target_link_libraries( VSProjectMakerCLI 
                 Qt5::Widgets
                 Qt5::Core
                 MainWindow
                 MainLib
                 SABUtils
          )
DeployQt( VSProjectMakerCLI .)
DeploySystem( VSProjectMakerCLI . )

INSTALL( TARGETS ${PROJECT_NAME} RUNTIME DESTINATION . )
INSTALL( FILES ${CMAKE_CURRENT_BINARY_DIR}/Debug/${PROJECT_NAME}.pdb DESTINATION . CONFIGURATIONS Debug RelWithDebInfo)
