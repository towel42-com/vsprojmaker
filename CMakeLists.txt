cmake_minimum_required(VERSION 3.1)
if(CMAKE_VERSION VERSION_LESS "3.7.0")
    set(CMAKE_INCLUDE_CURRENT_DIR ON)
endif()

project(root-VSProjectMaker) 

SET(CMAKE_MODULE_PATH ${CMAKE_SOURCE_DIR}/SABUtils/Modules ${CMAKE_MODULE_PATH})

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED true)
find_package(Threads)
find_package(Qt5 COMPONENTS Core Widgets REQUIRED)
find_package(Deploy REQUIRED)
find_package(AddUnitTest REQUIRED)
set_property( GLOBAL PROPERTY USE_FOLDERS ON )

SET( CMAKE_INSTALL_PREFIX C:/Users/sbloom/bin/VSProjMaker )

add_custom_target( SABTests_SABUtils )
add_subdirectory( SABUtils )
add_subdirectory( WizardPages )
add_subdirectory( MainWindow )
add_subdirectory( MainLib )
add_subdirectory( gui )
add_subdirectory( cli )

SET( CPACK_PACKAGE_VENDOR "Scott Aron Bloom" )
SET( CPACK_PACKAGE_VERSION_MAJOR "0" )
SET( CPACK_PACKAGE_VERSION_MINOR "22" )
SET( CPACK_PACKAGE_VERSION_PATCH "0" )
set( CPACK_NSIS_EXECUTABLES_DIRECTORY ".")
set( CPACK_PACKAGE_EXECUTABLES "VSProjectMakerGUI;Visual Studio Project Maker" )
set( CPACK_CREATE_DESKTOP_LINKS "VSProjectMakerGUI")
set(CPACK_NSIS_EXTRA_INSTALL_COMMANDS "CreateShortCut \\\"$DESKTOP\\\\Visual Studio Project Maker.lnk\\\" \\\"$INSTDIR\\\\VSProjectMakerGUI.exe\\\" ")

set( CPACK_NSIS_MODIFY_PATH OFF)
SET( CPACK_GENERATOR NSIS ZIP )
SET( CPACK_NSIS_MUI_ICON ${CMAKE_SOURCE_DIR}/MainWindow/resources/application.ico )
SET( CPACK_NSIS_MUI_UNIICON ${CMAKE_SOURCE_DIR}/MainWindow/resources/application.ico )
include( CPack )
